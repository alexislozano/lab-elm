Ports

port toJs : String -> Cmd msg

port toElm : (String -> msg) -> Sub msg

subscriptions model =
    toElm ReceiveFromJs

update msg model =
    case msg of
        SendToJs str ->
            ( model, toJs str )

        ReceiveFromJs str ->
            ( { model | message = str }, Cmd.none )
app.ports.toJs.subscribe(function (str) {
    app.ports.toElm.send("JS value");
});
